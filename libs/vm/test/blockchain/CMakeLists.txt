# Compiler Blockchain Tests

target_sources(compiler-blockchain-tests
    PRIVATE
        compiler_blockchain_tests.cpp
)

    PRIVATE
        "${TOP_CURRENT_BINARY_DIR}/test"
)

    blockchain-test-vm
    utils

    evmone-blockchaintest-lib)
